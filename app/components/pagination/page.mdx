# Pagination

The Pagination component enables the user to select a specific page from a range of pages.

## Basic Pagination
<Tabs items={['Preview', 'code']}>
  <Tabs.Tab value='Preview'>
    <BasicPagination />
  </Tabs.Tab>
  <Tabs.Tab value='Code'>
    ```jsx showLineNumbers
    import React from 'react';
    import Pagination from 'archi-ui/Pagination'

    export default function MyReactPage() {
      return (
        <div className='flex flex-col items-center justify-center gap-2'>
          <Pagination totalPages={10} />
          <Pagination totalPages={10} color='primary' />
          <Pagination totalPages={10} color='secondary' />
          <Pagination totalPages={10} disabled />
        </div>
      );
    }
    ```
  </Tabs.Tab>
</Tabs>

## Usage
```jsx
import Pagination from 'archi-ui/Pagination'
// or
import { Pagination } from 'archi-ui'
```
```jsx
<Pagination totalPages={10} />
```

## Controlled
If the `page` is set using useState, to update to next page or previous page on click, use `onPageChange()` to update the state of `page`.

<Tabs items={['Preview', 'code']}>
  <Tabs.Tab value='Preview'>
    <ControlledPagination />
  </Tabs.Tab>
  <Tabs.Tab value='Code'>
    ```jsx showLineNumbers
    import React, { useState } from 'react';
    import Pagination from 'archi-ui/Pagination'

    export default function MyReactPage() {

      const [page, setPage] = useState(3);

      const handlePageChange = (page) => {
        setPage(page)
      }

      return (
        <div className="flex items-center justify-center">
          <Pagination
            totalPages={8}
            variant='ghost'
            color='secondary'
            page={page}
            onPageChange={handlePageChange}
          />
        </div>
      );
    }
    ```
  </Tabs.Tab>
</Tabs>
