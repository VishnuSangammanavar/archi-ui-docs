# Switch

Switchs toggle the state of a single setting on or off

## Usage
```jsx
import Switch from 'archi-ui/Switch'
// or
import { Switch } from 'archi-ui'
```
```jsx
<Switch />
```

## Controlled
You can control the switch with the `checked` and `onChange` prop.

<Tabs items={['Preview', 'code']}>
  <Tabs.Tab value='Preview'>
    <ControlledSwitch />
  </Tabs.Tab>
  <Tabs.Tab value='Code'>
    ```jsx showLineNumbers
    "use client"

    import React, { useState }  from 'react'
    import Switch from 'archi-ui/Switch'

    export default function MyReactPage() {

      const [checked, setChecked] = useState(true)

      const handleChange = (event) => {
        setChecked(event.target.checked)
      }

      return (
        <div className="flex items-center justify-center gap-2">
          <Switch checked={checked} onChange={handleChange} variant='standard' />
          <p>Switch is checked: {String(checked)}</p>
        </div>
      );
    }
    ```
  </Tabs.Tab>
</Tabs>
